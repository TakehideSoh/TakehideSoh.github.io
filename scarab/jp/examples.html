Latin Square

 2: import jp.kobe_u.scarab.solver._
 3: import jp.kobe_u.scarab.sapp._
 4: 
 5: var n: Int = 5
 6: for (i <- 1 to n; j <- 1 to n)  int('x(i,j),1,n) 
 7: for (i <- 1 to n) {
 8:   add(alldiff((1 to n).map(j => 'x(i,j))))
 9:   add(alldiff((1 to n).map(j => 'x(j,i))))
10:   add(alldiff((1 to n).map(j => 'x(j,(i+j-1)%n+1))))
11:   add(alldiff((1 to n).map(j => 'x(j,(i+(j-1)*(n-1))%n+1))))}
12: 
13: if (find)  println(solution)
  • (Lines 1 to 3) import Scarab classes
  • (Line 6) declare integer variables
  • (Lines 8 to 11) add alldifferenct constraints for each row, column, and diagonal
  • (Line 12) print found solution if it exists

#When \(2n\) numbers \(\{1, 1, 2, 2, \ldots, n, n\}\) are given, (see also http://www.csplib.org/)

 4: 
 5: val n = 4
 6: 
 7: for (i <- 1 to 2*n) int('x(i),1,n)
 8: for (i <- 1 to n) 
 9:   add(Or(for (j <- 1 to 2*n-i-1) yield And(('x(j) === 'x(j+i+1)), ('x(j) === i))))

Graph Coloring

 1: import jp.kobe_u.scarab.csp._
 2: import jp.kobe_u.scarab.solver._
 3: import jp.kobe_u.scarab.sapp._
 4: 
 5: val nodes = Seq(1,2,3,4,5)
 6: val edges = Seq((1,2),(1,5),(2,3),(2,4),(3,4),(4,5))
 7: var maxColor = 4;
 8: 
 9: int('color,1,maxColor)
10: for (i <- nodes) int('n(i),1,maxColor)
11: for (i <- nodes) add('n(i) <= 'color)
12: for ((i,j) <- edges)  add('n(i) !== 'n(j))
13: 
14: while (find('color <= maxColor)) {
15:   println(solution)
16:   maxColor -= 1
17: }
  • (Lines 5 to 7) declare graph structure.
  • (Lines 9 and 10) declare integer variables representing available colors and each node of the given graph.
  • (Lines 11 and 12) declare constraints that limit available colors and adjacent nodes have different color.
  • (Lines 14 to 17) minimizing number of colors.

Alphametic Problem SAT + IS + FUN = TRUE

\(S*100 + A*10 + T + I*10 + S + F*100 + U*10 + N = T*1000 + R*100 + U*10 + E\) 

 1: import jp.kobe_u.scarab.csp._
 2: import jp.kobe_u.scarab.solver._
 3: import jp.kobe_u.scarab.sapp._
 4: 
 5: val base = 10
 6: 
 7: for (v <- Seq('s,'i,'f,'t)) yield int(v,1,base-1)     // S, I, F and T are not zero
 8: for (v <- Seq('a,'u,'n,'r,'e)) yield int(v,0,base-1)  // others can be zero
 9: for (v <- Seq('c1,'c2,'c3)) yield int(v,0,2)          // carries
10: 
11: add('t + 's + 'n       === 'e + 'c1*base)
12: add('a + 'i + 'u + 'c1 === 'u + 'c2*base)
13: add('s +      'f + 'c2 === 'r + 'c3*base)
14: add(               'c3 === 't)
15: 
16: add(alldiff(Seq('s,'i,'f,'t,'a,'u,'n,'r,'e)))
17: 
18: if (find)  println(solution.intMap)
  • (Lines 11 to 14) constraint model considering each digit and carry, which takes around 1 second;)

Open-shop Scheduling

 1: import jp.kobe_u.scarab.csp._
 2: import jp.kobe_u.scarab.solver._
 3: import jp.kobe_u.scarab.sapp._
 4: 
 5: use(new Sat4j("glucose"))
 6: 
 7: val pt = Seq(
 8:   Seq(661,   6, 333),
 9:   Seq(168, 489, 343),
10:   Seq(171, 505, 324))
11: 
12: val n = pt.size
13: val lb = pt.map(_.sum).max
14: var ub = (0 until n).map(k => (0 until n).map(i => pt(i)((i + k) % n)).max).sum
15: 
16: int('makespan, lb, ub)
17: 
18: for (i <- 0 until n; j <- 0 until n) {
19:   int('s(i,j), 0, ub)
20:   add('s(i,j) + pt(i)(j) <= 'makespan)
21: }
22: for (i <- 0 until n) {
23:   for (j <- 0 until n; l <- j+1 until n)
24:     add('s(i,j) + pt(i)(j) <= 's(i,l) ||
25:         's(i,l) + pt(i)(l) <= 's(i,j))
26: }
27: for (j <- 0 until n) {
28:   for (i <- 0 until n; k <- i+1 until n)
29:     add('s(i,j) + pt(i)(j) <= 's(k,j) ||
30:         's(k,j) + pt(k)(j) <= 's(i,j))
31: }
32: 
33: while (find('makespan <= ub)) {
34:   println(solution)
35:   val end = (for(i <- 0 until n; j <- 0 until n) 
36:              yield solution.intMap('s(i,j))+pt(i)(j)).max
37:   ub = end - 1
38:   println(ub)
39: }
  • (Lines 1 to 3) import Scarab classes
  • (Lines 7 to 10) declare an instance
  • (Lines 12 to 14) compute size, lower and upper bounds of the instance
  • (Line 16) declares an integer variable representing current makespan
  • (Lines 18 to 21) forces all operations are ended before makespan
  • (Lines 22 to 26) forces for operations in the same job do not overlap each other
  • (Lines 27 to 31) forces for operations sharing same resource do not overlap each other
  • (Lines 33 to 38) coumputes optimum solution

Colored N Queen

 1: import jp.kobe_u.scarab.csp._
 2: import jp.kobe_u.scarab.solver._
 3: import jp.kobe_u.scarab.sapp._
 4: 
 5: val n = args(0).toInt
 6: val c = n
 7: 
 8: use(new Sat4j("glucose"))
 9: 
10: for (i <- 1 to n; color <- 1 to c)
11:   int('q(i,color), 1, c)
12: 
13: for (color <- 1 to c) {
14:   add(alldiff((1 to n).map(i => 'q(i,color))))
15:   add(alldiff((1 to n).map(i => 'q(i,color)+i)))
16:   add(alldiff((1 to n).map(i => 'q(i,color)-i)))
17: }
18: 
19: for (i <- 1 to n)
20:   add(alldiff((1 to c).map(color => 'q(i,color))))
21: 
22: if (find) {
23:   for (color <- 1 to c) {
24:     for (row <- 1 to n) {
25:       var seq: Seq[Int] = Seq.empty
26:       for (column <- 1 to n)
27:   if (encoder.decode('q(row,color)) == column)
28:     seq = seq :+ color
29:   else
30:     seq = seq :+ 0
31:       println(seq.mkString(" "))
32:     }
33:     println("-----------------")
34:   }
35: }
  • (Lines 1 to 3) import Scarab classes
  • (Lines 5 to 6) size is given from command line
  • (Lines 8) declares the use of Sat4j of Glucose setting.
  • (Lines 10 to 11) declares integer variables representing queens
  • (Lines 13 to 17) representing N-Queen constraints for each color
  • (Lines 19 to 20) forces that Queens of each color do no overlap
  • (Lines 22 to 35) compute solutions and show the obtained placement

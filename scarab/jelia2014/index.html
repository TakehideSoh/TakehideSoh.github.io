 6: case class Graph(var nodes: Set[Int] = Set.empty, var edges: Set[(Int, Int)] = Set.empty) {
 7: 
 8:   def edge(n1: Int, n2: Int) = if (n1 < n2) (n1, n2) else (n2, n1)
 9: 
10:   private var adjacentMap: Map[Int, Set[Int]] = Map.empty
11:   private def addAdjacent(n1: Int, n2: Int) =
12:     adjacentMap += n1 -> (adjacentMap.getOrElse(n1, Set.empty) + n2)
13: 
14:   def addNode(n1: Int) = nodes += n1
15:   def addEdge(n1: Int, n2: Int) =
16:     if (n1 != n2) {
17:       edges += edge(n1, n2)
18:       addAdjacent(n1, n2)
19:       addAdjacent(n2, n1)
20:     }
21:   def adjacent(n: Int) = adjacentMap(n)
22:   def adjacentEdge(n: Int) = adjacent(n).map(n2 => edge(n, n2))
23: }

25: object Graph {
26:   def parse(source: Source): Graph = {
27:     val graph = Graph()
28:     val re = """e\s+(\d+)\s+(\d+)""".r
29:     for (line <- source.getLines.map(_.trim)) {
30:       line match {
31:         case re(s1, s2) => {
32:           val n1 = s1.toInt; graph.addNode(n1)
33:           val n2 = s2.toInt; graph.addNode(n2)
34:           graph.addEdge(n1, n2)
35:         }
36:         case _ =>
37:       }
38:     }
39:     graph
40:   }
41: }

43: def getCycle(node: Int, initial: Int, cycle: List[Int]): List[Int] = {
44:   val node2: Int = graph.adjacent(node).find(node2 => solver.solution('arc(node, node2)) > 0).get
45:   if (node2 == initial) node2 :: cycle
46:   else getCycle(node2, initial, node2 :: cycle)
47: }
48: 
49: def getCycles: Set[List[Int]] = {
50:   var cycles: Set[List[Int]] = Set.empty
51:   var nodes = graph.nodes
52:   while (!nodes.isEmpty) {
53:     val node = nodes.head
54:     val cycle = getCycle(node, node, List(node))
55:     cycles += cycle
56:     nodes --= cycle
57:   }
58:   cycles
59: }
60: 
61: def define = {
62:   for ((n1, n2) <- graph.edges) {
63:     int('arc(n1, n2), 0, 1)
64:     int('arc(n2, n1), 0, 1)
65:     add('arc(n1, n2) + 'arc(n2, n1) <= 1)
66:   }
67:   for (n1 <- graph.nodes) {
68:     val nodes = graph.adjacent(n1).toSeq
69:     add(Sum(nodes.map(i => 'arc(i, n1))) === 1)
70:     add(Sum(nodes.map(i => 'arc(n1, i))) === 1)
71:   }
72: }

74: def addBlockingClauses(cycle: List[Int]) {
75:   val ceArcs = for (edge <- cycle.sliding(2).toList) 
76:              yield (edge(0), edge(1))
77:   add(Or(ceArcs.map(i => 'arc(i._1, i._2) <= 0)))
78:   add(Or(ceArcs.map(i => 'arc(i._2, i._1) <= 0)))
79: }
